  6 /**
  7  * @class  Cette classe servira à manipuler les opérandes des opérations.
  8  * @constructor 
  9  * @argument {float} valeur c'est la valeur passer en paramêtre
 10  * @returns {Nombre}
 11  */
 12 function Nombre(valeur) {
 13     /**
 14      * @private
 15      * @type float
 16      */
 17     var valeur = valeur;
 18 
 19     if (isInt(valeur)) {
 20         /**
 21          * @private
 22          * @description text
 23          * @type Array
 24          */
 25         var partieEntiere = convertionStringEnTableau(String(valeur));
 26         /**
 27          * @private
 28          * @description text
 29          * @type Array
 30          */
 31         var partieDecimale = [];
 32 
 33     } else {
 34         var partieEntiere = convertionStringEnTableau(String(valeur).split(".")[0]);
 35         var partieDecimale = convertionStringEnTableau(String(valeur).split(".")[1]);
 36 
 37     }
 38     ;
 39     /**
 40      * @private
 41      * @description text
 42      * @type {integer}
 43      */
 44     var longueurValeur = partieEntiere.length + partieDecimale.length;
 45 
 46     /**
 47      * @public
 48      * @description text
 49      * @returns {Boolean}
 50      */
 51     this.isEmpty = function isEmpty() {
 52         if (valeur === null || valeur === "" || valeur === "undefined") {
 53             return true;
 54         } else {
 55             return false;
 56         }
 57     };
 58     /**
 59      * @public
 60      * @description text
 61      * @returns {float or int}
 62      */
 63     this.getValeur = function getValeur() {
 64         return valeur;
 65     };
 66     /**
 67      * @public
 68      * @description text
 69      * @returns {Array|tableau}
 70      */
 71     this.getPartieEntiere = function getPartieEntiere() {
 72         return partieEntiere;
 73     };
 74     /**
 75      * @public
 76      * @description text
 77      * @param {integer} indice
 78      * @returns {Array}
 79      */
 80     this.getPartieEntiereByIndice = function getPartieEntiereByIndice(indice) {
 81         if (partieEntiere[indice]) {
 82             return partieEntiere[indice];
 83         } else {
 84             return parseInt(0);
 85         }
 86     };
 87     /**
 88      * @public
 89      * @description text
 90      * @returns {Array}
 91      */
 92     this.getPartieDecimale = function getPartieDecimale() {
 93         return partieDecimale;
 94     };
 95     /**
 96      * @public
 97      * @description text
 98      * @param {integer} indice
 99      * @returns {Number}
100      */
101     this.getPartieDecimaleByIndice = function getPartieDecimaleByIndice(indice) {
102         if (partieDecimale[indice]) {
103             return partieDecimale[indice];
104         } else {
105             return parseInt(0);
106         }
107 
108     };
109     /**
110      * @public
111      * @description text
112      * @param {integer} valeur
113      * @returns {undefined}
114      */
115     this.setPartieEntiere = function setPartieEntiere(valeur) {
116         partieEntiere.push(valeur);
117     }
118     /**
119      * @public
120      * @description text
121      * @param {integer} indice
122      * @param {integer} valeur
123      * @returns {undefined}
124      */
125     this.setPartieEntiereByIndice = function setPartieEntiereByIndice(indice, valeur) {
126         if (partieEntiere[indice]) {
127             partieEntiere[indice] = valeur;
128         } else {
129             partieEntiere[indice] = parseInt(valeur);
130         }
131 
132     };
133     /**
134      * @public
135      * @description text
136      * @param {type} valeur
137      * @returns {undefined}
138      */
139     this.setPartieDecimale = function setPartieDecimale(valeur) {
140         partieDecimale.push(valeur);
141 
142     };
143     /**
144      * @public
145      * @description text
146      * @param {type} indice
147      * @param {type} valeur
148      * @returns {undefined}
149      */
150     this.setPartieDecimaleByIndice = function setPartieDecimaleByIndice(indice, valeur) {
151         if (partieDecimale[indice]) {
152             partieDecimale[indice] = parseInt(valeur);
153 
154         } else {
155             partieDecimale[indice] = parseInt(valeur);
156         }
157     };
158     /**
159      * 
160      * @returns {int}
161      */
162     this.getLongueurValeur = function getLongueurValeur() {
163         return longueurValeur;
164     };
165 
166     /**
167      * @public
168      * @description text
169      * @returns {String}
170      */
171     this.infos = function infos() {
172         return "valeur : " + valeur + "<br>" +
173                 "partieEntiere : " + partieEntiere + "<br>" +
174                 "partieDecimal : " + partieDecimale + "<br>" +
175                 "longueur : " + longueurValeur + "<br>";
176 
177     };
178 
179 }
180 /**
181  * @description Convertie une chaine de caractère en tableur
182  * @static
183  * @param {integer} valeur
184  * @returns {Array}
185  */
186 function convertionStringEnTableau(valeur) {
187 
188     var tableau = [];
189     var i;
190     for (i = 0; i < valeur.length; i++) {
191         tableau.push(parseInt(valeur[i]));
192     }
193 
194     return tableau;
195 }

197 /**
198  * @description Vérifie que le nombre est un entier
199  * @static
200  * @param {integer,float,string} nombre
201  * @returns {Boolean}
202  */
203 function isInt(nombre) {
204     if ((parseFloat(nombre) === parseInt(nombre)) && !isNaN(nombre)) {
205         return true;
206     } else {
207         return false;
208     }
209 }
