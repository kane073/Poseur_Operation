903         }

905             return {"intKeyCode": intKeyCode, "intAltKey": intAltKey, "intCtrlKey": intCtrlKey, "val": "del", "type": "action"};
906         }
907         if (expReg.test(caractere)) {
908             return {"intKeyCode": intKeyCode, "intAltKey": intAltKey, "intCtrlKey": intCtrlKey, "val": caractere, "type": "chiffre"};
909         }
910         if (caractere == "+" || caractere == "-" || caractere == "*" || caractere == "/" || caractere == ",") {
911             return {"intKeyCode": intKeyCode, "intAltKey": intAltKey, "intCtrlKey": intCtrlKey, "val": caractere, "type": "operateur"};
912         }
913         if (intKeyCode >= 37 && intKeyCode <= 40) {

926                     break;
927             }
928         }
929         //Touche tab

931             return {"intKeyCode": intKeyCode, "intAltKey": intAltKey, "intCtrlKey": intCtrlKey, "val": "droit", "type": "direction"};
932         }
933

935             return {"intKeyCode": intKeyCode, "intAltKey": intAltKey, "intCtrlKey": intCtrlKey, "val": caractere, "type": "nonpriseencharge"};
