  6 /**
  7  * @class  Cette classe servira à créer des opérations de type addition
  8  * @constructor 
  9  * @argument [Nombre,Nombre,...,Nombre] Elle prend en paramètre une liste d'élement type Nombre
 10  * @returns {Addition}
 11  */
 12 function Addition() {
 13     /**
 14      * 
 15      * @type arguments
 16      */
 17     var listArgument = arguments;
 18     /**
 19      * @private
 20      * @type Array
 21      */
 22     var operande = [];
 23     /**
 24      * @private
 25      * @type Array
 26      */
 27     var retenues = [0];
 28     /**
 29      * @private
 30      * @type Nombre
 31      */
 32     var resultat = new Nombre(0);
 33 
 34     /**
 35      * @private
 36      * @description Crèe des types Nombres et les mets dans l'atribut operande
 37      */
 38     function remplissageOperande() {
 39         if (listArgument.length > 0) {
 40             var i = 0;
 41             for (i = 0; i < listArgument.length; i++) {
 42                 operande.push(new Nombre(listArgument[i]));
 43             }
 44         } else {
 45             alert("Erreur, il n'y a pas d'argument!");
 46         }
 47     }
 48     ;
 49     remplissageOperande();
 50 
 51     /**
 52      * @private
 53      * @description Vérifie que le nombre d'argument passé en paramètre est compris entre 2 et 10
 54      * @returns {Boolean}
 55      */
 56     function verifieNumbreArgument() {
 57         if (listArgument.length >= 2 && listArgument.length <= 7) {
 58             return true;
 59         } else {
 60             alert("Le nombre d'opérande doit étre compris entre 2 et 10.");
 61             return false;
 62         }
 63     }
 64     ;
 65     /**
 66      * @private
 67      * @description Verifie que chaque argument est compris entre 1 et 18 chiffre
 68      * @returns {Boolean}
 69      */
 70     function verifieLongueurDeChaqueArgument() {
 71         var i = 0;
 72         var reponse = true;
 73         while (i < listArgument.length && reponse === true) {
 74             if (operande[i].getLongueurValeur() > 15 || operande[i].getLongueurValeur() < 1) {
 75                 reponse = false;
 76                 alert("L'un des arguments est supérieur à 15 caractères!");
 77             }
 78             i++;
 79         }
 80         return reponse;
 81     }
 82     ;
 83     /**
 84      * @private
 85      * @description 
 86      * @requires listArgument, operande
 87      * @returns {integer}
 88      */
 89     function longueurMaxPartieEntiere() {
 90         var longueurMax = 0;
 91         if (listArgument.length > 0) {
 92             for (i = 0; i < operande.length; i++) {
 93                 if (operande[i].getPartieEntiere().length > longueurMax) {
 94                     longueurMax = operande[i].getPartieEntiere().length;
 95                 }
 96             }
 97         }
 98         return longueurMax;
 99     }
100     /**
101      * @private
102      * @description 
103      * @requires listArgument, operande
104      * @returns {integer}
105      */
106     function longueurMaxPartieDecimale() {
107         var longueurMax = 0;
108         if (listArgument.length > 0) {
109             for (i = 0; i < operande.length; i++) {
110                 if (operande[i].getPartieDecimale().length >= longueurMax) {
111                     longueurMax = operande[i].getPartieDecimale().length;
112                 }
113             }
114         }
115         return longueurMax;
116     }
117     if (verifieLongueurDeChaqueArgument() && verifieNumbreArgument()) {
118 
119         this.isEmpty = function isEmpty() {
120 
121         };
122         /**
123          * @public
124          * @description text 
125          * @requires operande
126          */
127         this.resoudreAddition = function resoudreAddition() {
128             var MaxPartieEntiere = longueurMaxPartieEntiere();
129             var MaxPartieDecimale = longueurMaxPartieDecimale();
130             var matriceOprerande = [];
131 
132             for (i = 0; i < operande.length; i++) {
133                 var tmpPartieDecimal = [];
134                 if (operande[i].getPartieDecimale().length < MaxPartieDecimale) {
135                     for (j = 0; j < MaxPartieDecimale; j++) {
136                         if (j < operande[i].getPartieDecimale().length) {
137                             tmpPartieDecimal.push(parseInt(operande[i].getPartieDecimale()[j]));
138                         } else {
139                             tmpPartieDecimal.push(0);
140                         }
141                     }
142                 }
143                 else {
144                     tmpPartieDecimal = operande[i].getPartieDecimale();
145                 }
146                 var tmpPartieEntiere = [];
147                 if (operande[i].getPartieEntiere().length < MaxPartieEntiere) {
148                     for (j = 0; j < MaxPartieEntiere; j++) {
149                         if (j < operande[i].getPartieEntiere().length) {
150                             tmpPartieEntiere.push(parseInt(operande[i].getPartieEntiere()[j]));
151                         } else {
152                             tmpPartieEntiere.unshift(0);
153                         }
154                     }
155                 } else {
156                     tmpPartieEntiere = operande[i].getPartieEntiere();
157                 }
158                 matriceOprerande.push({entiere: tmpPartieEntiere, decimal: tmpPartieDecimal});
159             }
160 
161             var tmpSomme = [];
162             var tmp = 0;
163             for (j = MaxPartieDecimale - 1; j >= 0; j--) {
164                 tmp = 0;
165                 for (i = 0; i < matriceOprerande.length; i++) {
166                     tmp = tmp + matriceOprerande[i].decimal[j];
167                 }
168                 tmpSomme.unshift(tmp);
169             }
170             var taille;
171             var p;
172             var f;
173             var diff = 0;
174             for (i = tmpSomme.length - 1; i >= 0; i--) {
175                 taille = String(tmpSomme[i]).length;
176                 if (taille > 1) {
177                     p = String(tmpSomme[i])[taille - 1];
178                     f = String(tmpSomme[i]).substring(0, taille - 1);
179                     tmpSomme[i] = parseInt(p);
180                     if (i <= 0) {
181                         retenues.unshift(" ");
182                         tmpSomme.unshift(".");
183                         tmpSomme.unshift(parseInt(f));
184                         retenues.unshift(parseInt(f));
185                     } else {
186                         tmpSomme[i - 1] = tmpSomme[i - 1] + parseInt(f);
187                         retenues.unshift(parseInt(f));
188                     }
189                 } else {
190                     retenues.unshift(0);
191                 }
192             }
193             if (tmpSomme.length > 0) {
194                 diff = tmpSomme.length - 1;
195             } else {
196                 diff;
197             }
198 
199             var retenueDepart = 0;
200             if (retenues[0]) {
201                 retenueDepart = retenues[0];
202             }
203 
204             for (j = MaxPartieEntiere - 1; j >= 0; j--) {
205                 tmp = 0;
206                 for (i = 0; i < matriceOprerande.length; i++) {
207                     tmp = tmp + matriceOprerande[i].entiere[j];
208                 }
209                 if (j === MaxPartieEntiere - 1) {
210                     if (tmpSomme[0]) {
211                         tmpSomme[0] = tmp + retenueDepart;
212                     } else {
213                         tmpSomme.unshift(tmp + retenueDepart);
214                     }
215                 } else {
216                     tmpSomme.unshift(tmp);
217                 }
218             }
219 
220             for (i = tmpSomme.length - diff - 1; i >= 0; i--) {
221                 taille = String(tmpSomme[i]).length;
222                 if (taille > 1) {
223                     p = String(tmpSomme[i])[taille - 1];
224                     f = String(tmpSomme[i]).substring(0, taille - 1);
225                     tmpSomme[i] = parseInt(p);
226                     if (f.length <= 1) {
227                         if (i <= 0) {
228                             tmpSomme.unshift(parseInt(f));
229                             retenues.unshift(parseInt(f));
230                         } else {
231                             tmpSomme[i - 1] = tmpSomme[i - 1] + parseInt(f);
232                             retenues.unshift(parseInt(f));
233                         }
234                     } else {
235                         sf = f.substring(0, f.length - 1);
236                         if (i <= 0) {
237                             tmpSomme.unshift(parseInt(f));
238                             retenues.unshift(parseInt(f));
239                             if (f.length > 1) {
240                                 for (z = 0; z < f.length; z++) {
241                                     if (tmpSomme[i]) {
242                                         tmpSomme[i] = parseInt(f[z + f.length - 1]);
243                                     } else {
244                                         tmpSomme.unshift(parseInt(f[f.length - z - 1]));
245                                         retenues.unshift(parseInt(f[f.length - z - 1]));
246                                     }
247                                 }
248                             }
249                         } else {
250                             tmpSomme[i - 1] = tmpSomme[i - 1] + parseInt(f);
251                             retenues.unshift(parseInt(f));
252                         }
253                     }
254                 } else {
255                     if (i !== 0) {
256                         retenues.unshift(0);
257                     }
258                 }
259             }
260 
261             var valCreateResultat = "";
262             for (i = 0; i < tmpSomme.length; i++) {
263                 valCreateResultat += String(tmpSomme[i]);
264             }
265             resultat = new Nombre(parseFloat(valCreateResultat));
266 
267         };
268 
269         /**
270          * @public
271          * @description text
272          * @returns {Nombre}
273          */
274         this.getResultat = function getResultat() {
275             return resultat;
276         };
277         /**
278          * @public
279          * @description text
280          * @returns {Array}
281          */
282         this.getRetenues = function getRetenues() {
283             return retenues;
284         };
285         /**
286          * @public
287          * @description text
288          * @returns {Array}
289          */
290         this.getOperande = function getOperande() {
291             return operande;
292         };
293 
294     }
295 }
